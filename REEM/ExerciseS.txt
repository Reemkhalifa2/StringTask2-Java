Exercise 1: Simple Calculator 

START
  BEGIN
    DISPLAY ENTER NUMBER 1
    READ NUMBER1
    SET NUM1 TO NUMBER1
    DISPLAY ENTER NUMBER 2
    READ NUMBER2
    SET NUM2 TO NUMBER2
    SET SUM TO NUM1 + NUM2
    DISPLAY SUM
  STOP
END

Exercise 2: Pass or Fail 

START
  BEGIN
    DISPLAY ENTER YOUR SCORE
    READ SCORE
    IF SCORE GREATER THAN OR EQUAL TO 50 THEN
      DISPLAY PASS
    ELSE
      DISPALY FAIL
    END IF
  STOP
END

Exercise 3: Coffee Shop Order

END

Exercise 4: Even or Odd

START
  BEGIN
    DISPLAY ENTER NUMBER
    READ NUMBER
    IF NUMBER MOD 2 EQUALS 0 THEN
      DISPALY EVEN NUMBER
    ELSE
      DISPLAY ODD NUMBER
    END IF
  STOP
END

Exercise 5: Count to 10 

START
  BEGIN
    SET I TO 1
    WHILE I LESS THAN OR EQUAL TO 10
      DISPLAY I
      SET I TO I+1
    END WHILE
  STOP
END

Exercise 6: Discount Calculator 

START
  BEGIN
    DISPLAY ENTER THE PRICE
    READ PRICE

    IF PRICE GREATER THAN OR EQUAL TO 100 THEN
       DISPLAY PRICE - PRICE * 0.10
    ELSE
       DISPALY PRICE
    END IF 
  STOP
END

Exercise 8: Largest of Three Numbers
START
  BEGIN
    DISPLAY READ NUM1
    READ NUM1
    DISPLAY READ NUM2
    READ NUM2
    DISPLAY READ NUM3
    READ NUM3
    IF NUM1 IS GREATER THAN NUM2 AND NUM1 IS GREATER THAN NUM3 THEN
      SET LARGEST TO NUM1
    ELSE IF NUM2 IS GREATER THAN NUM1 AND NUM2 IS GREATER THAN NUM3 THEN
      SET LARGEST TO NUM2
    ELSE
      SET LARGEST TO NUM3
    END IF
    DISPLAY The largest number is + + LARGEST
  STOP 
END
    

Exercise 9: Multiplication Table

END

Exercise 10: Grade Calculator 

START
  BEGIN
    READ SCORE
    IF SCORE GREATER THAN OR EQUALS 90 THEN
     DISPLAY Your grade is A
    ELSE IF SCORE GREATER THAN OR EQUALS 80 THEN
     DISPLAY Your grade is B
    ELSE IF SCORE GREATER THAN OR EQUALS 70 THEN
     DISPLAY Your grade is C
    ELSE IF SCORE GREATER THAN OR EQUALS 60 THEN
     DISPLAY Your grade is D
    ELSE 
     DISPLAY Your grade is F
    END IF 
SOTP 
END

Exercise 12: Positive, Negative, or Zero

START
  BEGIN
    DISPLAY ENTER NUMBER
    READ NUM
    IF NUM GREATER THAN 0 THEN
      DISPLAY POSITIVE
    ELSE IF NUM LESS THAN 0 THEN
      DISPLAY NEGATIVE
    ELSE
      DISPLAY Zero
    END IF
  STOP
END
