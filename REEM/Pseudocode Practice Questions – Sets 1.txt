
-------------------- Q114 — No Duplicates Check ---------------------------------


START
	BEGIN
		SET FRUITS TO LIST["apple", "banana", "orange"]
		CREATE SET noDuplicates
		REPEAT FROM I = 1 TO SIZE OF FRUITS
      			ADD FRUITS[I] TO SET noDuplicates
   		END REPEAT
		IF SIZE OF FRUITS EQUALS SET SIZE OF noDuplicates
		THEN
			DISPLAY TRUE
		ELSE
			DISPLAY FALSE
		END IF

	END
STOP


--------------------- Q115 — Shared Members --------------------------------

START
	BEGIN
		SET A TO LIST["Ali", "Sara"]
		SET B TO LIST["Tom", "Sara"]

		CREATE SET SETA
		CREATE SET SETB

		REPEAT FROM I = 1 TO SIZE OF A
			ADD A[I] TO SET SETA
		END REPEAT

		REPEAT FROM I = 1 TO SIZE OF B
			ADD B[I] TO SET SETB
		END REPEAT

		CREATE SET sharedMembers
		SET sharedMembers TO INTERSECT SETS SETA AND SETB

		IF SET SIZE OF sharedMembers GREATER THAN 0 THEN
			DISPLAY TRUE
		ELSE
			DISPLAY FALSE
		END IF
	END
STOP

---------------------- Q118 — Remove Banned Words -----------------------------------



START
	BEGIN
		SET words TO LIST["hi", "bad", "hello", "evil"]
		SET removeBanned TO LIST[]

		CREATE SET banned_words
		ADD "bad" TO SET banned_words
		ADD "evil" TO SET banned_words

		REPEAT FROM I = 1 TO SIZE OF words
			IF NOT SET CONTAINS banned_words words[I] THEN
				ADD words[I] TO LIST removeBanned
			END IF
		END REPEAT

		DISPLAY removeBanned
	END
STOPP

------------------------- Q122 — Symmetric Difference --------------------


START
	BEGIN

		CREATE SET SETA
		ADD "a" TO SET SETA
		ADD "b" TO SET SETA
		ADD "c" TO SET SETA

		CREATE SET SETB
		ADD "b" TO SET SETB
		ADD "c" TO SET SETB
		ADD "d" TO SET SETB

		SET BOTH_SETS TO INTERSECT SETS SETA AND SETB
		SET MERGESETS TO MERGE SETS SETA AND SETB

		REPEATE FROM I=1 TO SET SIZE OF BOTH_SETS
			REMOVE FROM SET MERGESETS BOTH_SETS[I]
		END REPEATE
			

		DISPLAY MERGESETS

	END
STOP




----------------------- Q123 — First Repeated Value  --------------------



START
	BEGIN
		SET TEXT TO LIST["a", "b", "c", "b", "d"]
		CREATE SET firstRepeated
		REPEAT FROM I = 1 TO SIZE OF TEXT
			IF SET CONTAINS firstRepeated TEXT[I] THEN
				DISPLAY TEXT[I]
				
			ELSE
				ADD TEXT[I] TO SET firstRepeated
			END IF
		END REPEAT

	END
STOP



--------------- Q124 — Unique Voter Check ------------------------


START
	BEGIN
		CREATE SET voterCheck
		ADD "ID001" TO SET voterCheck
		ADD "ID002" TO SET voterCheck

		SET VOTE_ID TO "ID001"

		IF SET CONTAINS voterCheck VOTE_ID THEN
			DISPLAY "Already Voted"
		ELSE
			ADD VOTE_ID SET voterCheck
			DISPLAY "Vote Accepted"
		END IF
	END
STOP


------------------- Q125 — Tag Merger  --------------------------


START
	BEGIN
