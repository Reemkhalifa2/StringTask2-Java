---------------   TASK 1 — Age Category Checker---------------

CLASS PERSON
	
	ATTRIBUTES
	name : TEXT
	age : NUMBER
	METHODS
	CHECK_CATEGORY() 
		IF AGE LESS THAN 13 THEN
			DISPLAY name + "- Child"
		ELSE IF AGE LESS THAN OR EQUAL TO 19 THEN
			DISPLAY name + "- Teenager"
		ELSE
			DISPLAY name + "- Adult"
		END IF
	END METHOD
END CLASS


START
	BEGIN
		CREATE OBJECT newPerson OF CLASS PERSON
		SET newPerson.name TO "Sara"
		SET newPerson.age TO 15
		SET newPerson.CHECK_CATEGORY()
	END
STOP

------------ TASK 3 — Wallet System  -----------------


CLASS WALLET
	ATTRIBUTES
	owner : TEXT
	money : NUMBER
	METHODS
	ADD_MONEY(amount)
		SET money TO money + amount
		DISPLAY "Money:" + money
	END METHOD
	SPEND_MONEY(amount)
	IF amount GREATER THAN money THEN
		DISPLAY  "Not enough money" 
	ELSE
		SET money TO money - amount
	END IF
	DISPLAY "Money:" + money
	END METHOD
END CLASS

START
	BEGIN 
		CREATE OBJECT person1 OF WALLET
		SET person1.owner to "REEM"
		SET person1.money to 100
		person1.ADD_MONEY(50)
		person1.SPEND_MONEY(30)
		person1.SPEND_MONEY(40)
	END 
STOP

TASK 17 — Exam Pass Counter ----------------------------------

CLASS Exam
	ATTRIBUTES
	marks : LIST 
	METHODS
	COUNT_PASS()
	SET PASS TO 0 
	REPEAT FROM I=1 TO SIZE OF marks
		IF marks[I] GREATER THAN OR EQUAL TO 50 THEN
			SET PASS TO PASS + 1
		END IF
	END REPEAT
	DISPLAY "Passed students: "+ PASS
	END METHOD
END CLASS

START
	BEGIN
		CREATE OBJECT midExam OF Exam
		SET midExam.marks TO LIST[30,60,80,45]
		midExam.COUNT_PASS()
	END 
STOP
